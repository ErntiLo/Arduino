#include <Servo.h>
Servo servom;
int but1=2;
int but2=3;
int but3=4;
int but4=5;

void setup()
{
  pinMode(but1, INPUT);
  pinMode(but2, INPUT);
  pinMode(but3, INPUT);
  pinMode(but4, INPUT);
  servom.attach(13);
}

void loop()
{
  
  if(digitalRead(but1)==1)
  {
   
   servom.write(45); 
  }
   if(digitalRead(but2)==1)
  {
   servom.write(90); 
  }
   if(digitalRead(but3)==1)
  {
   servom.write(135); 
  } if(digitalRead(but4)==1)
  {
   servom.write(180); 
  }
  
}
